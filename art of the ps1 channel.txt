import requests

# Image URL (public domain / Wikimedia)
url = "https://www.brainyquote.com/photos_tr/en/b/benjaminfranklin/383997/benjaminfranklin1-2x.jpg"

# Output file name
output_file = "Sample.jpg"

try:
    response = requests.get(url, stream=True)
    response.raise_for_status()  # Raise error for bad status

    with open(output_file, "wb") as file:
        for chunk in response.iter_content(1024):
            file.write(chunk)

    print(f"Image successfully downloaded and saved as {output_file}")

except requests.exceptions.RequestException as e:
    print(f"Error downloading image: {e}")
